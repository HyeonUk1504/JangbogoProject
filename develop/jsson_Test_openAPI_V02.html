def concat_read_file(arg_fnm_list):
    global dir_path
    fd_path = dir_path

    dfs = pd.DataFrame()
    for fnm in arg_fnm_list:
        df = pd.read_excel(fd_path+fnm, skiprows=[0, 1, 2, 3, 5], thousands=',') # 파일을 열었을때 무시하는 행
        dfs = pd.concat([dfs,df], join='outer', ignore_index=True)
    return dfs
